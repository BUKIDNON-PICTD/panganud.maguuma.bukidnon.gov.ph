import { Component, OnInit, ViewChild } from '@angular/core';
import { IonInfiniteScroll } from '@ionic/angular';
import { Socket } from 'ng-socket-io';
import { environment } from '../../../environments/environment';

@Component({
  selector: 'app-farmerslist',
  templateUrl: './farmerslist.page.html',
  styleUrls: ['./farmerslist.page.scss'],
})
export class FarmerslistPage implements OnInit {

  @ViewChild(IonInfiniteScroll) infiniteScroll: IonInfiniteScroll;
  public dataList: any;

  public start: number;
  public limit: number;
  public rowsize: number;
  public searchtext = '';
  constructor(public socket: Socket) {
    this.dataList = [];
    this.rowsize = 25;
    this.start = 0;
    this.limit = 25;
    //sample request
    this.socket.emit('serverrequest', {
      reciever : 'bukidnon',
      sender : environment.clientcode,
      servicename	: 'FarmerProfileService',
      methodname	: 'getFarmersList',
      paging 		: {
        start : this.start,
        limit : this.limit
      },
      searchtext: this.searchtext
    });
    //sample response
    this.socket.on('serverresponse', (data) => {
      // console.log(data);
      data.forEach(element => {
        element.no =  this.dataList.length;
        this.dataList.push(element);
      });
    });
  }

  loadData(event) {
    setTimeout(() => {
      console.log('Done');
       //sample request
      this.start = this.start + this.rowsize;
      this.socket.emit('serverrequest', {
        reciever : 'bukidnon',
        sender : environment.clientcode,
        servicename	: 'FarmerProfileService',
        methodname	: 'getFarmersList',
        paging 		: {
          start : this.start,
          limit : this.limit
        },
        searchtext: this.searchtext

      });
      
      event.target.complete();
      // App logic to determine if all data is loaded
      // and disable the infinite scroll
      if (this.dataList.length === 1000) {
        event.target.disabled = true;
      }
    }, 500);
  }
  setFilteredItems() {
    this.dataList = [];
    this.socket.emit('serverrequest', {
      reciever : 'bukidnon',
      sender : environment.clientcode,
      servicename	: 'FarmerProfileService',
      methodname	: 'getFarmersList',
      paging 		: {
        start : 0,
        limit : 25
      },
      searchtext: this.searchtext
    });
  }
  toggleInfiniteScroll() {
    this.infiniteScroll.disabled = !this.infiniteScroll.disabled;
  }

  ngOnInit() {
    
  }
}
